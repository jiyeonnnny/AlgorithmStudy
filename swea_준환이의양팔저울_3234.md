def next_permutation(lst):
    i = n-1
    while (i>0 and lst[i-1]>=lst[i]):
        i-=1
    if i==0:
        return False
    j=n-1
    while (lst[i-1]>=lst[j]):
        j-=1
    lst[i-1], lst[j] = lst[j], lst[i-1]
    k = n-1
    while (i<k):
        lst[i],lst[k] = lst[k],lst[i]
        i+=1
        k-=1
    return True

for t in range(int(input())):
    n = int(input())
    arr = list(map(int, input().split()))
    total = sum(arr)
    l = r = 0

    ans = 0
    stack = []
    for i in range(n):
        stack.append([arr[i],0,(1<<i)])

    while stack:
        l, r, visit = stack.pop()

        if l<r:
            continue

        if visit == ((1 << n) - 1):
            print(l,r,visit)
            ans+=1
            continue

        for i in range(n):
            if (1<<i) & visit==0:
                nxt_visit = ((1<<i)|visit)
                if l+arr[i]>=r:
                    stack.append([l+arr[i], r, nxt_visit])
                elif l>=r+arr[i]:
                    stack.append([l, r+arr[i], nxt_visit])
