            dfs(nx,ny)

def bfs():
    q = deque([[0,0]])
    while q:
        x,y = q.popleft()
        for dx, dy in [[0,1],[0,-1],[1,0],[-1,0]]:
            nx = x + dx
            ny = y + dy
            if 0<=nx<n and 0<=ny<m:
                if arr[nx][ny] not in d:
                    d.add(arr[nx][ny])
                    q.append([nx,ny])

n, m = map(int, input().split())
arr = [list(input().strip()) for _ in range(n)]
d = {arr[0][0]}
ans = 0
bfs()
print(len(d))
