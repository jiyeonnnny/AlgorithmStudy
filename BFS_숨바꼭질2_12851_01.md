def bfs():
    global k,n

    q = deque([n])
    while q:

        x = q.popleft()
        if dp[k][0]>0 and dp[x][0] > dp[k][0]:
            return

        for dx in [2*x, x+1, x-1]:
            if 0<=dx<=100000:
                if dx==k:
                    dp[dx][0] = dp[x][0] + 1
                    dp[k][1] += dp[x][1]

                elif dp[dx][0]==-1:
                    dp[dx][0] = dp[x][0] + 1 # time
                    dp[dx][1] += dp[x][1]
                    q.append(dx)

import sys
input = sys.stdin.readline
from collections import deque

def bfs():
    q = deque([n])
    cnt = 1
    while q:
        x = q.popleft()
        if dp[k]!=-1 and dp[x] > dp[k]:
            return cnt

        for nx in [2*x, x+1, x-1]:
            if 0 <= nx <= 100000:
                if nx == k:
                    dp[nx] = dp[x] + 1
                    cnt += 1
                elif dp[nx]== -1:
                    dp[nx] = dp[x] + 1
                    q.append(nx)
                elif dp[nx]==dp[x]+1:
                    q.append(nx)

n,k = map(int, input().split())
dp = [-1]*100001
dp[n]=0
if n==k:
    print(0)
    print(1)
else:
    cnt = bfs()
    print(dp[k])
    print(cnt-1)

    while q:

            cnt+=1

        for nx in [2*x, x+1, x-1]:
            if 0 <= nx <= 100000 and (dp[nx]==-1 or dp[nx]==dp[x]+1):
